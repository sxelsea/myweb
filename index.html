// Класс «Комната общежития»
class DormRoom {
  int maxCapacity;
  int currentCapacity = 0;
  List students = [];

  DormRoom(this.maxCapacity);

  void addStudent(Student student) {
    if (student.name.toLowerCase() == 'комендант') {
      students.add(student);
      currentCapacity++;
      print("${student.name} добавлен(-а) в комнату. Комендант может находиться в комнате в любое время");
    } else {
      if (currentCapacity >= maxCapacity) {
        print("Ошибка: Превышен максимальный размер комнаты. Невозможно добавить ${student.name}");
      } else {
        students.add(student);
        currentCapacity++;
        print('${student.name} добавлен(-а) в комнату');
      }
    }
  }

  @override
  String toString() {
    return 'Комната общежития:\n'
           'Максимальная вместимость: $maxCapacity\n'
           'Текущая вместимость: ${students.join(', ')}';
  }
}

void main() {
  var dormRoom = DormRoom(3);

  dormRoom.addStudent(Student('Леонардо'));
  dormRoom.addStudent(Student('Анастасия'));
  dormRoom.addStudent(Student('Саша'));
  dormRoom.addStudent(Student('Комендант'));

  print(dormRoom);

  dormRoom.addStudent(Student('Алекс'));
  print(dormRoom);
}

// 2. Реализовать класс Vehicle и классы-наследники Car и Bicycle
abstract class Vehicle {
  void move();
}

class Car extends Vehicle {
  @override
  void move() {
    print('Автомобиль едет по дороге.');
  }
}

class Bicycle extends Vehicle {
  @override
  void move() {
    print('Велосипед едет по тропинке.');
  }
}

void main() {
  Vehicle myCar = Car();
  myCar.move();

  Vehicle myBicycle = Bicycle();
  myBicycle.move();
}
