import 'dart:io';

void main() {
    print("Введите номер месяца (1-12):");

    int month = int.parse(stdin.readLineSync()!);

    switch(month) {
        case 1:
        case 3:
        case 5:
        case 7:
        case 8:
        case 10:
        case 12:
            print("В этом месяце 31 день");
            break;
        case 4:
        case 6:
        case 9:
        case 11:
            print("В этом месяце 30 дней");
            break;
        case 2:
            print("В этом месяце 28 дней");
            break;
        default:
            print("Вы ввели некорректный номер месяца.");
    }
}

import 'dart:io';

int findMax(List numbers) {
    int maxNumber = numbers[0];
    for (int i = 1; i  maxNumber){
            maxNumber= numbers[0];
        }
    }
    return maxNumber;
}
void main() {
    print("Введите список чисел через пробел:");
    List numbers = stdin.readLineSync()!.split(' ').map(int.parse).toList();

    int maxValue = findMax(numbers);
    print("Максимальное число из списка: $maxValue");
}

import 'dart:io';
void main() {
    print("Введите список чисел:");
    List numbers = stdin.readLineSync()!.split(' ').map(int.parse).toList();

    int maxValue = numbers.reduce((a, b) => a > b ? a : b);
    print("Максимальное значение в списке: $maxValue");
}

Function createCounter() {
  int count = 0;

  return () {
    count++;
    return count;
  };
}

void main() {
  var counter = createCounter();

  print(counter()); 
  print(counter());
