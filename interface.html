// abstract class Animal {
void eat() => print("Животное ест.");
void sleep() => print("Животное спит.");
}

mixin CanFly { void fly() => print("Летает."); }
mixin CanSwim { void swim() => print("Плавает."); }
mixin CanRun { void run() => print("Бегает."); }

class Bird extends Animal with CanFly {}
class Fish extends Animal with CanSwim {}
class Dog extends Animal with CanRun {}

void main() {
  Bird().fly();
  Fish().swim();
  Dog().run();
}

Задание 2

// abstract class Shape {
double getArea();
}
 
class Circle implements Shape {
double radius;
 
Circle(this.radius);
 
@Override
double getArea() {
  return 3.14 * radius * radius;
}
}
 
class Square implements Shape {
double side;
 
Square(this.side);
 
@Override
double getArea() {
  return side * side;
}
}
 
void main() {
Shape circle = Circle(5);
Shape square = Square(4);
 
print("Circle area: ${circle.getArea()}");
print("Square area: ${square.getArea()}");
}

Задание 3

// mixin SoundMixin {
void makeSound() {
  print("Making sound...");
}
}
 
mixin RunMixin {
void run() {
  print("Running...");
}
}
 
class Dog with SoundMixin, RunMixin {
void display() {
  print("I am a dog");
}
}
 
void main() {
Dog dog = Dog();
dog.display();
dog.makeSound();
dog.run();
}
